140 inline void Advance()
141 {
142 const TickType new_ticks = tick_service_->Milliseconds();
143 const TickType delta = current_ticks_ ? new_ticks - current_ticks_ : 0;
144 current_ticks_ = new_ticks;
145
146 if (delta == 0) {
147 return;
148 }
149
150 if (delta >= static_cast<TickType>(total_buckets_)) {
151 Clear();
152 return;
153 }
154
155 for (TickType i = 0; i < delta; ++i) {
156 auto& bucket = buckets_[(bucket_index_ + i) % total_buckets_];
157 for (const auto& key : bucket) {
158 cache_.erase(key);
159 }
160 bucket.clear();
161 }
162
163 bucket_index_ = (bucket_index_ + delta) % total_buckets_;
164
165 return;
166 }

168 template<typename Value>
169 inline void InternalInsert(const K& key, Value value, size_t ttl)
170 {
171 if (ttl > duration_) {
172 throw std::invalid_argument("TTL is greater than max duration");
173 } else if (ttl == 0) {
174 ttl = duration_;
175 }
176
177 ttl = ttl / interval_;
178
179 Advance();
180 const IndexType future_index = (bucket_index_ + ttl - 1) % total_buckets_;
181
182 buckets_[future_index].push_back(key);
183 cache_[key] = std::make_shared<T>(value);
184 }
