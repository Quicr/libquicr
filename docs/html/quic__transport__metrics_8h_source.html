14 struct MinMaxAvg
15 {
16 uint64_t min{ 0 };
17 uint64_t max{ 0 };
18 uint64_t avg{ 0 };
19
20 uint64_t value_sum{ 0 };
21 uint64_t value_count{ 0 };
22
23#if __cplusplus >= 202002L
24 constexpr auto operator<=>(const MinMaxAvg&) const = default;
25#else
26 constexpr bool operator==(const MinMaxAvg& other) const
27 {
28 return min == other.min && max == other.max && avg == other.avg && value_sum == other.value_sum;
29 }
30 constexpr bool operator!=(const MinMaxAvg& other) const { return !(*this == other); }
31#endif
32
40 void AddValue(const uint64_t value)
41 {
42 min = min ? std::min(min, value) : value;
43 max = std::max(max, value);
44
45 value_sum += value;
46 value_count++;
47
48 if (!value_count)
49 value_count = 1;
50
51 avg = value_sum / value_count;
52 }
53
54 void Clear()
55 {
56 min = 0;
57 max = 0;
58 avg = 0;
59 value_sum = 0;
60 value_count = 0;
61 }
62 };
