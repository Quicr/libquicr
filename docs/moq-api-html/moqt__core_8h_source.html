56 struct TrackHash
57 {
58 j
59 uint64_t track_namespace_hash; // 64bit hash of namespace
60 uint64_t track_name_hash; // 64bit hash of name
61
62 uint64_t track_fullname_hash; // 62bit of namespace+name
63
64 TrackHash(const uint64_t name_space, const uint64_t name)
65 {
66 track_namespace_hash = name_space;
67 track_name_hash = name;
68 }
69
70 TrackHash(const FullTrackName& tfn) noexcept
71 {
72 track_namespace_hash = std::hash<std::string_view>{}(
73 { reinterpret_cast<const char*>(tfn.track_namespace.data()), tfn.track_namespace.size() });
74 track_name_hash =
75 std::hash<std::string_view>{}({ reinterpret_cast<const char*>(tfn.name.data()), tfn.name.size() });
76
77 track_fullname_hash = (track_namespace_hash ^ (track_name_hash << 1)) << 1 >>
78 2; // combine and convert to 62 bits for uintVar
79 }
