75 struct TrackHash
76 {
77 uint64_t track_namespace_hash; // 64bit hash of namespace
78 uint64_t track_name_hash; // 64bit hash of name
79
80 uint64_t track_fullname_hash; // 62bit of namespace+name
81
82 TrackHash(const uint64_t name_space, const uint64_t name) {
83 track_namespace_hash = name_space;
84 track_name_hash = name;
85 }
86
87 TrackHash(const TrackFullName& tfn) noexcept
88 {
89 track_namespace_hash = std::hash<std::string_view>{}(
90 {reinterpret_cast<const char*>(tfn.name_space.data()), tfn.name_space.size()});
91 track_name_hash = std::hash<std::string_view>{}(
92 {reinterpret_cast<const char*>(tfn.name.data()), tfn.name.size()});
93
94 track_fullname_hash = (track_namespace_hash ^ (track_name_hash << 1)) << 1 >> 2; // combine and convert to 62 bits for uintVar
95 }
96 };
