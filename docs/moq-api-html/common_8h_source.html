77
78 TrackHash(const FullTrackName& ftn) noexcept
79 {
80 track_namespace_hash = std::hash<std::string_view>{}(
81 { reinterpret_cast<const char*>(ftn.name_space.data()), ftn.name_space.size() });
82 track_name_hash =
83 std::hash<std::string_view>{}({ reinterpret_cast<const char*>(ftn.name.data()), ftn.name.size() });
84
85 track_fullname_hash =
86 (track_namespace_hash ^ (track_name_hash << 1)) << 1 >> 2; // combine and convert to 62 bits for uintVar
87 }
88 };
