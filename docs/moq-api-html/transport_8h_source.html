56 struct TrackHash
57 {
58 uint64_t track_namespace_hash; // 64bit hash of namespace
59 uint64_t track_name_hash; // 64bit hash of name
60
61 uint64_t track_fullname_hash; // 62bit of namespace+name
62
63 TrackHash(const uint64_t name_space, const uint64_t name)
64 {
65 track_namespace_hash = name_space;
66 track_name_hash = name;
67 }
68
69 TrackHash(const FullTrackName& tfn) noexcept
70 {
71 track_namespace_hash = std::hash<std::string_view>{}(
72 { reinterpret_cast<const char*>(tfn.track_namespace.data()), tfn.track_namespace.size() });
73 track_name_hash =
74 std::hash<std::string_view>{}({ reinterpret_cast<const char*>(tfn.name.data()), tfn.name.size() });
75
76 track_fullname_hash = (track_namespace_hash ^ (track_name_hash << 1)) << 1 >>
77 2; // combine and convert to 62 bits for uintVar
78 }
79 };
