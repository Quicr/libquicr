55 struct TrackHash
56 {
57 uint64_t track_namespace_hash; // 64bit hash of namespace
58 uint64_t track_name_hash; // 64bit hash of name
59
60 uint64_t track_fullname_hash; // 62bit of namespace+name
61
62 TrackHash(const uint64_t name_space, const uint64_t name)
63 {
64 track_namespace_hash = name_space;
65 track_name_hash = name;
66 }
67
68 TrackHash(const FullTrackName& ftn) noexcept
69 {
70 track_namespace_hash = std::hash<std::string_view>{}(
71 { reinterpret_cast<const char*>(ftn.name_space.data()), ftn.name_space.size() });
72 track_name_hash =
73 std::hash<std::string_view>{}({ reinterpret_cast<const char*>(ftn.name.data()), ftn.name.size() });
74
75 track_fullname_hash = (track_namespace_hash ^ (track_name_hash << 1)) << 1 >>
76 2; // combine and convert to 62 bits for uintVar
77 }
78 };
